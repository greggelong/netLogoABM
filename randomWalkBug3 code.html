turtles-own [message?] ; true or false? does a turthe own a message


to setup ; in order to set up a new model, do the following
  clear-all
  reset-ticks
  create-turtles 75;
  [
    set message? false ; nobody has message
    setxy random-xcor random-ycor
  ]
  ask turtles
  [
    set shape "bug" ; check out others under tools turtle shape editor
    set size 1
    set color green
  ]
  ask one-of turtles [set message? true] ; give one of the turtles a message
end


to go ; in order to "go get moving do the following"
  ask turtles
  [
    ifelse coin-flip? [right random 60] [left random 60] 
    ; if coin flip is true (0)
    forward random 4 ; between 0 and 3 steps
    ; if the turtle meets another turtle 
    if any? other turtles-here with [message?]; same patch 
    [set message? true];
    if message? [set color yellow] ; sets turtle with message to yellow 
  ]
  tick ; add one unit of time to counter
end

to-report coin-flip?
  report random 2 = 0
  
end
